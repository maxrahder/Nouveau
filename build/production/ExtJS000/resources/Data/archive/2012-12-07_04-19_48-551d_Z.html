Internally, the 5 types are simply calling Ext.Msg.show() with user-supplied and hard-coded configs.

	progress : function(cfg, msg, progressText){
	       if (Ext.isString(cfg)) {
	           cfg = {
	               title: cfg,
	               msg: msg,
	               progress: true,
	               progressText: progressText
	           };
	       }
	       return this.show(cfg);
	   }
