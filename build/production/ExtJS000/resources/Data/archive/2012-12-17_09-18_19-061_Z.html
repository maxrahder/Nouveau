• You can extend / inherit from other CSS classes
  • Improves readability & maintainability
SCSS
.alert {
    font-size: 0.9em;
    background-color: yellow;
}

.alert .warning {
    font-weight: bold;
}

.redalert {
    @extend .alert;
    background-color: red;
}
