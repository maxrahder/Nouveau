• The entry point of your app, execution of your app starts
  • Use Ext.app.Application instead of Ext.onReady in an MVC-based application
  • Provides dependencies to Controllers, Models, Stores and Views
  • Use autoCreateViewport to automatically load and instantiate AppName.view.Viewport before firing the launch method
Ext.Loader.setPath({
    'MusicLibrary': 'app',
});

Ext.application({
    controllers: ["Main"],
    name: 'MusicLibrary', //this name will be your namespace
    autoCreateViewport: true,
    launch: function(){
      console.log("Everything is ready, let's execute the code.");
    };
});
