Ext.Array.forEach(array, fn, [scope])

Iterates an array and invoke the given callback function for each item.

It doesn't support stopping the iteration by returning false in the callback function like Ext.Array.each().

var a = [{value: 1}, {value: -1}, {value: 3}, {value : 99}, {value: 3}];
var s = '';
Ext.Array.forEach(a, function(item){
    if (item.value === 99) return false;
    if (item.value > 0){
        s += "-" + item.value;
    } 
});
alert(s);  // alerts -1-3-3
