The items config array specifies the child components. 

Ext.create('Ext.container.Container', {

    layout : 'vbox',	
    items : [ {
        xtype : 'button',
        text : 'Button One',
        margin : 8
    }, {
        xtype : 'button',
        text : 'Button Two',
        margin : 8
    } ],

    renderTo : Ext.getBody()
});
