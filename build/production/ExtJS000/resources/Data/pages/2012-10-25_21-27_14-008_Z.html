This chart has one series plotted against two axes

Ext.define('MyApp.model.Sales', {
    extend : 'Ext.data.Model',
    fields : [ 'name', {
        name : 'sales',
        type : 'int'
    } ]
});
store = Ext.create('Ext.data.Store', {
    model: 'MyApp.model.Sales',
    proxy : {
        type : 'ajax',
        url : '../Data/charts/musicianSales.json'
    },
    autoLoad : true
});

chart = Ext.create('Ext.chart.Chart', {
    store : store,
    axes : [ {
        type : 'Numeric',
        position : 'left',
        fields : [ 'sales' ],
        title : 'Sales ($millions)',
        minimum : 0
    }, {
        type : 'Category',
        position : 'bottom',
        fields : [ 'name' ],
        title : 'Artist or Band'
    } ],
    series : [ {
        type : 'column',
        highlight : true,
        xField : 'name',
        yField : 'sales'
    } ]
});

Ext.create('Ext.container.Container', {
    renderTo: Ext.getBody(),
    layout : 'fit',
    height: 350,
    items : [ chart ]
});
