Ext.Array.forEach() visit every element in an array. It calls the browser's native forEach() method, if there is one.

var a = [{value: 1}, {value: 2}, {value: 3}];
var s = '';
Ext.Array.forEach(a, function(item, index, array){
    s += item.value + ' ';
});
