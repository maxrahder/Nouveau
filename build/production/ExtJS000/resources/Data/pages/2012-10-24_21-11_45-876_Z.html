Each element in the fields array is an Ext.data.Field instance. Fields can be typed and given a default value. 

Ext.define('MyApp.model.Person', {
    extend : 'Ext.data.Model',
    fields : [ {
        name : 'first',
        type : 'string',
        defaultValue : 'John'
    }, {
        name : 'last',
        type : 'string',
        defaultValue : 'Doe'
    }, {
        name : 'dob',
        type : 'date',
        dateFormat : 'Y/m/d'
    } ]
});
var record = Ext.create('MyApp.model.Person');
alert(record.get('first')); // Alerts "John"
record.set('dob', '1955/05/23'); // Set one value

record.set({first : 'Jane', dob : '1955/05/23'}); //Set multiple values
alert(record.get('first')); // Alerts "Jane"
