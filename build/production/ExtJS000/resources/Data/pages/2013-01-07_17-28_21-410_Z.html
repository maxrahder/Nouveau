When using Ext.create(), pass a single string or multiple strings. With the tpl property pass an object, a single string or an array of strings. 

var c = Ext.create('Ext.Component', {
    tpl: [
        '<h1>', 
        '{name}', 
        '</h1>', 
        '<h2>', 
        '{age}', 
        '</h2>' 
    ],
    //tpl : Ext.create('Ext.XTemplate','<h1>{name}</h1><h2>{age}</h2>'),
    //tpl: '<h1>{name}</h1><h2>{age}</h2>',
    renderTo : Ext.getBody()
});
c.update({ name: 'Homer Simpson', age: 38 });
