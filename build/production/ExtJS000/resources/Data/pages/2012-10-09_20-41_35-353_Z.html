Panels also have a buttons array, used to show buttons at the bottom right of the panel. Note that Ext.window.Window is a panel subclass, so this feature might be useful in a dialog box containing a form.

This panel has two items, and in addition, two buttons in the buttons array.

Under the covers, the buttons array is implemented as a bottom docked toolbar with special styling. 

var w = Ext.create('Ext.window.Window', {
    title : 'Person',
    layout : 'fit',
    items : {
        xtype : 'form',
        bodyPadding : 8,
        items : [ {
            xtype : 'textfield',
            fieldLabel : 'First'
        }, {
            xtype : 'textfield',
            fieldLabel : 'Last'
        }, {
            xtype : 'datefield',
            fieldLabel : 'Birth Date'
        } ]
    },
    buttons : [ {
        text : 'Save',
        handler : function(b) {
            b.up('window').close();
        }
    }, {
        text : 'Cancel',
        handler : function(b) {
            b.up('window').close();
        }
    } ]
});
w.showAt(20, 20);
