In addition to passing strings to the template being created, you can also pass an object containing functions. These can then be referenced from within the template. 

var tpl = Ext.create('Ext.XTemplate', (
    '...',
    {
        isFemale: function(gender){
            return (gender === 'female');
        },
        isKid: function(age){
            return (age < 18);
        }
    }
);
