The default column type is xtype:'gridcolumn'. It simply runs toString() on the record's value.

var store = Ext.create('Ext.data.Store', {
    fields:[ { name: 'born',  type: 'date' , dateFormat : 'Y/m/d'}, 'name' ],
    proxy : { type : 'ajax', url : '../Data/json/marxBrothers.json' },
    autoLoad : true
});

Ext.create('Ext.grid.Panel', {

    store: store,
    columns: [ { 
        //xtype : 'gridcolumn', // This is the default
        text: 'Name', 
        dataIndex: 'name' 
    }, { 
        text: 'Born', 
        dataIndex: 'born',
        flex: 1 
    } ],

    title: 'The Marx Brothers',
    height: 200,
    width: 400,
    renderTo: Ext.getBody()
});
