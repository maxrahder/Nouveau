Alert Read-only message

Ext.Msg.alert('Widget Updated', 'The widget was updated successfully.');

Confirm Verifies an action

Ext.Msg.confirm('Are you sure?', 'Are you sure you want to delete this widget?', function(btn, text){
	if(btn === 'yes'){
    	Ext.Msg.alert('Deleted', 'The widget you selected has been deleted');
	}
	else{
		Ext.Msg.alert('Not Deleted', 'The widget you selected was not deleted');
	}
});

Prompt Asks user for text input

Ext.Msg.prompt('Name', 'Please enter your name:', function(btn, text){
	var name = (btn === 'ok' && text.length) ? text : 'anonymous user';
    Ext.Msg.alert('Hello', 'Hello, ' + name + '!');
});

Progress Progress bar indicator

Ext.Msg.progress('Processing', 'Please wait while we process your application...');

setTimeout(function(){ Ext.Msg.updateProgress(0.2, '20% complete') }, 1000);
setTimeout(function(){ Ext.Msg.updateProgress(0.42, '42% complete') }, 3000);
setTimeout(function(){ Ext.Msg.updateProgress(0.73, '73% complete') }, 3500);
setTimeout(function(){ Ext.Msg.updateProgress(1, 'Complete') }, 4000);

Wait Indefinite progress bar indicator

Ext.Msg.wait('Please wait while we process your application...', 'Processing');
