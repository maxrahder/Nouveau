Ext.Array.some(array, fn, [scope]):boolean

Returns boolean whether some element matches some rule."

// Return from begin index through end index-1
var somethingMatches; 
var a = ['john', 'paul', 'ringo'];

somethingMatches = Ext.Array.some(a, function(item){
    return (item === 'paul');
});
alert(somethingMatches); // Alerts true

somethingMatches = Ext.Array.some(a, function(item){
    return (item === 'george');
});
alert(somethingMatches); // Alerts false
