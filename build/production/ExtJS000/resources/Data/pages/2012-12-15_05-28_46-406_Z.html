• If you set the tree panel's rootVisible property to false, you just see the root's children.
Ext.create('Ext.tree.Panel', {
    rootVisible: false,
    root: {
        text: 'Root Node',
        expanded: true,
        children: [{
            text: 'Child 1',
            expanded: true,
            children: [{
                text: 'Child 1.1',
                leaf: true
            }, {
                text: 'Child 1.2',
                leaf: true
            }]
        }, {
            text: 'Child 2',
            expanded: true,
            children: [{
                text: 'Child 2.1',
                leaf: true
            }]
        }]
    },
    title: 'Tree With Children, (rootVisible: false)',
    renderTo: Ext.getBody()
});
