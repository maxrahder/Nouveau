xtype : 'splitbutton' Ext.button.Split

Ext.button.Split subclasses Ext.button.Button.

Split buttons are split, so that the drop-down icon is separate from the rest of the button. This lets you have the main button have an action without having to show the sub-items. 

Ext.create('Ext.panel.Panel', {
    title: ' ',
    renderTo: Ext.getBody(),
    height: 200,
    bodyPadding: 16,
    items: [{
        xtype: 'splitbutton',
        text: 'Choose an Action',
        width: 120,
        handler: function(b) {
            if (b.sound) {
                var h = b.up('panel').down('header');
                h.setTitle(h.title + ' ' + b.sound);
            }
        },
        menu: {
            defaults: {
                handler: function(b) {
                    var parent = b.up('splitbutton');
                    parent.setText(b.text);
                    parent.sound = b.sound;
                }
            },
            items: [{
                text: 'Sing',
                sound: 'la la la'
            }, {
                text: 'Play Guitar',
                sound: '<i>strum</i>'
            }, {
                text: 'Compose',
                sound: '<i>scribble</i>'
            }]
        }
    }]
});
