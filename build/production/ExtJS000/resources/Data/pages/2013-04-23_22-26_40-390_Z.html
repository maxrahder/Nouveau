Ext.define('MyApp.model.Person', {
    extend: 'Ext.data.Model',
    fields: ['first', 'last'],
    proxy: {
        type: 'ajax',
        url: '../Data/json/theBeatles.json'
    }
});
Ext.create('Ext.data.Store', {
    model: 'MyApp.model.Person',
    autoLoad: true,
    storeId: 'People'
});
Ext.create('Ext.grid.Panel', {
    store: 'People',
    columns: [{
        dataIndex: 'first',
        text: 'First'
    }, {
        dataIndex: 'last',
        flex: 1,
        text: 'Last'
    }],
    width: 300,
    renderTo: Ext.getBody()
});
