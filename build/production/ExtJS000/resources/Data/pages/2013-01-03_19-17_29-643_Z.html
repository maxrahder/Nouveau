Controller instances automatically have getters corresponding to every item in the stores, models and views arrays. 

Ext.define('MyApp.controller.BeatlesController', {
    extend : 'Ext.app.Controller',
    views : [ 'BeatlesGrid' ],
    stores : [ 'People' ],
    models : [ 'Person' ],
    myMethod : function(){

        // This is handy --  same as Ext.getStore('People')
        this.getPeopleStore();

        // These aren't so useful
        this.getBeatlesGridView(); // The grid class constructor
        this.getPersonModel();     // The model class constructor
    }
});
