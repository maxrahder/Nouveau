each(iterable, fn, [scope], [reverse])

Iterates an array or an iterable value and invoke the given callback function for each item.

It can break a loop with return false

var a = [{value: 1}, {value: -1}, {value: 3}, {value : 99}, {value: 3}];
var s = '';
Ext.Array.each(a, function(item){
    if (item.value === 99) return false;
    if (item.value > 0){
        s += "-" + item.value;
    } 
});
alert(s);  // alerts -1-3
