Ext.application() has built-in MVC and dynamic loading features.

Ext.application({
    name: 'MyApp',
    requires: ['Ext.Viewport', 'Ext.button.Button'],
    controllers: ['MyController'],
    launch: function(){
        Ext.create('Ext.Viewport', {
            items: [{
                xtype: 'button',
                text: 'Hi There!',
                margin: 32
            }]
        });
    }
});
