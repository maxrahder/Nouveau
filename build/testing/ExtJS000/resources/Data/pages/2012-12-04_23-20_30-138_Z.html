Ext.JSON is a singleton used to convert an object to a JSON string, or a JSON string to an object. It's commonly used when using Ext.Ajax to send data, or 

var jsonString = "{a : 1, b : [ { c : 'hi' } ] , d: function(){alert('In the function')}}";

var o = Ext.JSON.decode(jsonString); // Turn it into an object

console.log(o); // Show the new object
o.d(); // Run the function

o.a = '932487293'; // Update the object
delete o.d; // Remove a property

var s = Ext.JSON.encode(o); // Turn it back into a string
alert(s);
