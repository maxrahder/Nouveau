There is an optional third parameter used with on(). It it establishes the value of "this" within the event handler.

var b = Ext.create('Ext.button.Button', {
    text : 'Button',
    renderTo : Ext.getBody()
});
b.on('click', 
    function(button){
        alert(this.getText() + ' click');
    }, 
    b
);
