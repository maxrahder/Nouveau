• A tree store creates it's own behind-the-scenes model if you do not define one.

  • A model in this case defines the fields for the each of the nodes (rows) in the tree.

  • Ext JS adds properties from the Ext.data.NodeInterface class to the node model. These properties include text, children, expanded, and leaf.

var aTreeStore = Ext.create('Ext.data.TreeStore', {
    root: {
        expanded: true,
        children: [{
            text: 'Node 1',
            expanded: true,
            children: [{
                text: 'Node 1.1',
                leaf: true
            }, {
                text: 'Node 1.2',
                leaf: true
            }]
        }, {
            text: 'Node 2',
            expanded: true,
            children: [{
                text: 'Node 2.1',
                leaf: true
            }]
        }]
    }
});

Ext.create('Ext.tree.Panel', {
    store: aTreeStore,
    title: 'Tree With TreeStore',
    renderTo: Ext.getBody()
});
