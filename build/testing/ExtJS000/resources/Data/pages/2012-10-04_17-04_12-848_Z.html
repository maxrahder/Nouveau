Send and receive asynchronous data.

  • Same domain via AJAX
  • Cross domain via JSONP
Ext.Ajax.request({
    url: 'php/twitter/twitterProxy.php?query=extjs',
    success: function(response) {
        var resp = Ext.JSON.decode(response.responseText);
        t.overwrite(Ext.getBody(), resp.statuses);
    }
});
