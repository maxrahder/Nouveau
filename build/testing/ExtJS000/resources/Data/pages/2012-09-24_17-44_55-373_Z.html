The defaults config lets you set default values for direct child items. 

Ext.create('Ext.container.Container', {

    defaults: {
        xtype: 'button',
        style: 'background: red',
        margin: 8
    },
    layout : 'vbox',	
    items : [ {
        text : 'Button One'
    }, {
        text : 'Button Two'
    }, {
        xtype: 'container',
        items: [
           { xtype: 'button' , text: 'Button A' }, 
           { xtype: 'button' , text: 'Button B' }
        ]
    }],

    renderTo : Ext.getBody()
});
