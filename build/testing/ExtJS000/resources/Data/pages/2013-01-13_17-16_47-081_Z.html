Ext.define('Associations.model.Customer', {
    extend : 'Ext.data.Model',
    fields : [ 'id', 'name', 'phone' ],


    hasMany : {
        model : 'Associations.model.Order',
        name : 'orders',
        foreignKey : 'customer_id'
    },


    proxy : {
        type : 'ajax',
        url : 'customer.php'
    },
});
