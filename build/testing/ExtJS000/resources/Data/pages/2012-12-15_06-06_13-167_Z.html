• Each node may have an array of children.

Ext.create('Ext.tree.Panel', {
    root: {
        text: 'Root Node',
        expanded: true,
        children: [{
            text: 'Child1',
            leaf: true
        }, {
            text: 'Child2',
            leaf: true
        }]
    },
    title: 'Tree With Root and 2 Child Nodes',
    renderTo: Ext.getBody()
});
