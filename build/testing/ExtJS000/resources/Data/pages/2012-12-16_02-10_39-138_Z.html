• You can load the data in from a file using a proxy.
  • In this case, the json file contains an object with a children property, and an array of child objects as a value.
{  "children": [{
        "text": "Topic 1",
        "expanded": true,
        "children": [{
            "text": "Sub 1.1",
            "leaf": true
        }, {
            "text": "Sub 1.2",
            "leaf": true
        }]
    },
    {
        "text": "Topic 2",
        "expanded": true,
        "children": [{
            "text": "Sub 2.1",
            "leaf": true
        }, {
            "text": "Sub 2.2",
            "leaf": true
        }]
    }]
}
  • The store still uses a root object unless the tree panel's rootVisible is false.
var aTreeStore = Ext.create('Ext.data.TreeStore', {
    root: {
        text: 'Root Node',
        expanded: true
        },
    proxy: {
        type: 'ajax',
        url: '../Data/json/treeExample.json'
    }
});

Ext.create('Ext.tree.Panel', {
    store: aTreeStore,
    title: 'Tree With Proxy',
    renderTo: Ext.getBody(),
});
