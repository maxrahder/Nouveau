Ext.Function.createThrottled(fn, interval, [scope]):Function

Creates a throttled version of the passed function. When called, the original function is run immediately, but upon subsequent calls, only runs after the interval has passed.

var handler = function(button){
    Ext.DomHelper.append(Ext.getBody(), "<b><large>&nbsp;Hi</large></b>");
};
var throttledHander = Ext.Function.createThrottled(handler, 1000);

Ext.create('Ext.button.Button', {
    text: 'Hi',
    renderTo: Ext.getBody(),
    clicks : 0,
    handler: function(button){
        throttledHander(button);
    }
});
