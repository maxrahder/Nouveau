The following basic math operators may be applied directly on numeric data values: + - * /

//array with data
var data = {
    name: 'The Simpsons',
    family: [
        { name: 'Homer',  age: 38, gender: 'male' },
        { name: 'Margareth',  age: 36, gender: 'female' },
        { name: 'Bart',    age: 10, gender: 'male' },
        { name: 'Lisa',   age: 8, gender: 'female' },
        { name: 'Maggie', age: 1, gender: 'female' }
    ]
};

//template
var tpl = new Ext.XTemplate(
    '<h2>Family: {name}</h2><br>',
    '<table class="table">',
    '<tpl for="family">',
        '<tpl if="this.isKid(age)">',
            '<tr><td colspan="2">{name} is still a kid.',
             '<p>{name} is to young! {age - 18} years...</p>',
            '</td>',
        '</tpl>',
    '</tpl>',
    '</table>',
    {
        // XTemplate configuration:
        disableFormats: true,
        // member functions:
        isKid: function(age){
           return age < 18;
        }
    }
);
tpl.overwrite(Ext.getBody(), data);
