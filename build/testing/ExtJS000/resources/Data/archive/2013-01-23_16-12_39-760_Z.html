var members = Ext.create('Ext.data.Store', {
    autoLoad : true,
    fields: ['first', 'last', {name: 'dob', type: 'date', dateFormat: 'Y/m/d'}],
    proxy: { type: 'ajax', url: '../Data/json/aerosmith.json' }
});

Ext.create('Ext.grid.Panel',{
    store: members,
    selModel : {
        selType : 'rowmodel',
        mode : 'MULTI'
    },
    columns: [{
        text: 'First Name', 
        dataIndex: 'first'
    }, {
        text: 'Last Name',
        dataIndex: 'last'
    }, {
        xtype : 'datecolumn',
        text: 'Born',
        format : 'F j, Y',
        dataIndex: 'dob',
        flex : 1
    }],
    title: 'Aerosmith',
    height: 200,
    width: 450,
    renderTo: Ext.getBody()
});
