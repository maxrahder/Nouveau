• Easily change color contrasts.
SCSS
$red: #FF0000;
$blue: #0000FF;
.dark {
   background: darken($red, 20%);
}

.light {
   background: lighten($red, 20%);
}

.color-mix {
   background: $red + $blue;
}

.alpha {
  background: transparentize($red, 0.25);
}
