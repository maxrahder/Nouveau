Ext.define('Mammal', {
	config : {
		name : 'Anonymous',
		dob : new Date()
	},
	constructor : function(config) {
		this.initConfig(config);
		this.callParent(config);
	},
	speak : function() {
		alert(this.getName() 
			+ ' is grunting.');
	}
});
   
Ext.define('Human', {
	extend : 'Mammal',
	config : {
		married : false
	},
	constructor : function(config) {
		this.initConfig(config);
		this.callParent(config);
	},
	speak : function() {
		alert('Hello there, my name is ' 
			+ this.getName() 
			+ '. I am ' 
			+ (this.getMarried() ? 'not ' : '')
			+ 'married.');
	}
});

var m = Ext.create('Mammal', {
	name : 'Big Foot'
});
// Alerts "Big Foot is grunting."
m.speak();
  
var h = Ext.create('Human', {
	name : 'Hubert',
	married : true
});
// Alerts "Hello there, my name is Hubert. I am married."
h.speak();
