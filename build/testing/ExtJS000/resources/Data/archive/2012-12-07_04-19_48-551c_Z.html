Internally, the 5 types are simply calling Ext.Msg.show() with user-supplied and hard-coded configs.

	prompt : function(cfg, msg, fn, scope, multiline, value){
	       if (Ext.isString(cfg)) {
	           cfg = {
	               prompt: true,
	               title: cfg,
	               minWidth: this.minPromptWidth,
	               msg: msg,
	               buttons: this.OKCANCEL,
	               callback: fn,
	               scope: scope,
	               multiline: multiline,
	               value: value
	           };
	       }
	       return this.show(cfg);
	   }
