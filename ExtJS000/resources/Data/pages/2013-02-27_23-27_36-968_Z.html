Open the debugger and inspect the panel.

  • Create the button: The component now exists, but there's no DOM.
  • Add the button to the panel: The button renders and you see it in the DOM.
  • Remove the button: This removes it from the DOM (but the component still exists).
var panel = Ext.create('Ext.panel.Panel', {
    tbar : [{
        text : 'Create Button',
        handler : function(){
            panel.button = panel.button || Ext.create('Ext.button.Button', {text : 'BUTTON', itemId : 'mybutton'});
        }
    }, {
        text : 'Add Button to Panel',
        handler : function(){
            panel.add(panel.button);
        }
    }, {
        text : 'Remove Button from Panel',
        handler : function(){
            panel.remove(panel.button, false); // false: Don't delete the component
        }
    }],
    height : 100,
    width : 400,
    bodyPadding : 8,
    renderTo : Ext.getBody()
});
