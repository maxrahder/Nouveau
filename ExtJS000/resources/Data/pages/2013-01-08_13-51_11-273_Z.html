<tpl for=".">...</tpl> loops over the elements of an array. The "$" symbol contains the property name. The "." symbol is the property value.

var data = { 
    name: 'Homer Simpson',  
    age: 38 
};

var tpl = Ext.create('Ext.XTemplate',
    '<tpl foreach=".">',
        '<p> {$} : {.} </p>',
    '</tpl>'
);
tpl.overwrite(Ext.getBody(), data);
