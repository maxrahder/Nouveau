If you call any function via new, the function is called in the context of a new object. 

var Person = function(name){
	this.name = name;
	this.speak = function(){
		alert('Hello, my name is ' + this.name);
	}
	return this;
}

// The following two statements do the same thing.
var p = new Person('Hubert');
// var p = Person.call({}, 'Hubert');  

alert(p.name); // alerts 'Hubert'
p.speak(); // alerts 'Hello, my name is Hubert'
