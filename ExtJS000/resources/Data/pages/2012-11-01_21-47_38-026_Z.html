Ext.Number.randomInt(from, to):Number

Returns a random integer, in the specified range.

var s;

s = '';
for (var i = 0 ; i < 50 ; i++){
    s += Ext.Number.randomInt(0, 100) + ' '; 
}
alert(s);

s = '';
for (var i = 0 ; i < 10 ; i++){
    // To illustrate that it's inclusive  
    s += Ext.Number.randomInt(0, 1) + ' '; 
}
alert(s);
