The following example illustrates the use of custom error messages:

  { 
  xtype: 'textfield',
  name: 'firstName',
  fieldLabel: 'First Name',
  allowBlank: false,
  blankText: 'Please enter your first name',
  minLength: 2,
  minLengthText: 'You must enter at least two characters.',
  maxLength: 20,
  maxLengthText: 'Your entered name must be between 2 and 20 characters'
 }
