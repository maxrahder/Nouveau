Bar charts are unusual in that the independent variable is on the x axis

var bearData =

[ { type : 'Polar'  , weight : 600, speed : 40 }, 
  { type : 'Black'  , weight : 270, speed : 48 }, 
  {type : 'Grizzly' , weight : 225, speed : 56 } ];

var store = Ext.create('Ext.data.Store', {
    fields : ['type', 'weight', 'speed'],
    data : bearData
})

Ext.create('Ext.chart.Chart', {
    renderTo : Ext.getBody(),
    width : 500,
    height : 300,
    animate : true,
    store : store,
    axes : [ {
        type : 'Numeric',
        position : 'bottom',
        minimum : 35,
        maximum : 60,
        majorTickSteps : 4,
        title : 'Speed (kph)'
    }, {
        type : 'Category',
        position : 'left',
        fields : [ 'type' ],
        title : 'Type of Bear'
    } ],
    series : [ {
        type : 'bar',
        axis : [ 'bottom', 'left' ],
        xField : 'type',
        yField : 'speed'
    } ]
});
