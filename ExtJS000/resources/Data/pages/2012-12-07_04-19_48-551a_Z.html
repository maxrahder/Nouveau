Internally, the 5 types are simply calling Ext.Msg.show() with a predetermined set of configs.

alert: function(cfg, msg, fn, scope) {
  if (Ext.isString(cfg)) {
     cfg = {
         title : cfg,
         msg : msg,
         buttons: this.OK,
         fn: fn,
         scope : scope,
         minWidth: this.minWidth
     };
  }
  return this.show(cfg);
}
