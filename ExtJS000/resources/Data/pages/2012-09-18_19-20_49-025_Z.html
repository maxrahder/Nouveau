You can define your own convert(value, record) function for any column. You can use this to define additional fields, not in the data feed.

Ext.define('MyApp.model.Person', {
    extend : 'Ext.data.Model',
    fields : [ 'first', 
    { name : 'last', type : 'string' },

    { name : 'full',
      convert : function(value, record) {
          return record.get('first') + ' ' + record.get('last');
      }
    }, 

    { name : 'dob', type : 'date', dateFormat : 'Y/m/d' }]
});

var record = Ext.create('MyApp.model.Person', {
   first : 'John', 
   last : 'Lennon', 
   dob : '1940/10/09'
});

Ext.Msg.alert('Beatle', record.get('full'));
