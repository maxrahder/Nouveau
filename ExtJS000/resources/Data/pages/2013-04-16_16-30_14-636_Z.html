The menu config can also be a full Ext.menu.Menu config. 

Ext.create('Ext.panel.Panel', {
    title : 'Panel',
    renderTo : Ext.getBody(),
    height : 200,
    bodyPadding : 16,
    items : [{
        xtype : 'button',
        text : 'Beatles',
        menu : {
            defaults : {
                handler: function(b){
                    var adjectives = ['fab', 'gear', 'groovy', 'hip', 'dreamy'];
                    var i = Ext.Number.randomInt(0, (adjectives.length - 1));
                    alert(b.text + ' is ' + adjectives[i] + '!');
                }
            },
            items : [{
                text : 'John'
            }, {
                text : 'Paul'
            }, {
                text : 'George'
            }, {
                text : 'Ringo'
            }]
        }
    }]
});
