• A tree panel has several properties that can determine its look and feel.

  • lines turns the appearance of lines between the nodes on and off. Default is true.

  • useArrows adds triangles to the left of the tree. Default is false.

  • animate animates node expansion/collapsing. Default is the value of Ext.enableFx, which by default is true.

Ext.create('Ext.tree.Panel', {
    lines: false,
    useArrows: true,
    animate: false,
    root: {
        text: 'Root Node',
        expanded: true,
        children: [{
            text: 'Child 1',
            expanded: true,
            children: [{
                text: 'Child 1.1',
                leaf: true
            }, {
                text: 'Child 1.2',
                leaf: true
            }]
        }, {
            text: 'Child 2',
            expanded: true,
            children: [{
                text: 'Child 2.1',
                leaf: true
            }]
        }]
    },
    title: 'Tree With Triangles and No Lines',
    renderTo: Ext.getBody()
});
