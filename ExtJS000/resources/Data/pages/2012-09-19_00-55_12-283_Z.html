We can also do visual effects

Ext.getBody().update('<p id="myParagraph">Hi there</p>');
Ext.fly('myParagraph').highlight("#ff0000", { 
	attr: 'backgroundColor', 
	duration: 2000 
});

Ext.getBody().update('<p id="myParagraph">Hi there</p>');
Ext.fly('myParagraph').fadeOut({duration: 2000});
Ext.fly('myParagraph').fadeIn({duration: 2000});
