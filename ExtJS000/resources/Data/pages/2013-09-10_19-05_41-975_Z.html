It's up to you how you implement localization in your code. Here's the approach used by Ext JS: 

Ext.define('MyApp.view.NameForm',{
    extend: 'Ext.form.Panel',
    firstNameLabel: 'First Name',
    lastNameLabel: 'Last Name',
    initComponent: function(){
        this.items = [{
            name: 'name',
            fieldLabel: this.firstNameLabel
        }, {
            name: 'last',
            fieldLabel: this.lastNameLabel
        }];
        this.callParent();
    }
});
Ext.define('MyApp.locale.fr.view.NameForm',{
    override: 'MyApp.view.NameForm',
    firstNameLabel: 'Pr√©nom',
    lastNameLabel: 'Nom de Famille'
});
