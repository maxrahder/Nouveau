• It is common to create an Ext.data.TreeStore to contain the data for the tree.
  • The properties for the root object remain the same.
  • Note that as changes happen in the tree store, the tree will update automatically.
var aTreeStore = Ext.create('Ext.data.TreeStore', {
    root: {
        text: 'Root Node',
        expanded: true,
        children: [{
            text: 'Child 1',
            expanded: true,
            children: [{
                text: 'Child 1.1',
                leaf: true
            }, {
                text: 'Child 1.2',
                leaf: true
            }]
        }, {
            text: 'Child 2',
            expanded: true,
            children: [{
                text: 'Child 2.1',
                leaf: true
            }]
        }]
    }
});

Ext.create('Ext.tree.Panel', {
    store: aTreeStore,
    title: 'Tree With TreeStore',
    renderTo: Ext.getBody()
});
