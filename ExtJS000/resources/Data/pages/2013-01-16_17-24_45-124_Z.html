Now the designer is free to change how the search is initiated, without breaking the controller. 

{
    xtype : 'textfield',
    itemId : 'searchText',
    fieldLabel : 'Search',
    enableKeyEvents : true,
    listeners : {
        keypress : function(field, event) {
            if (event.getKey() === event.ENTER) {
                field.up('twittergrid').fireSearch();
            }
        }
    }
}
