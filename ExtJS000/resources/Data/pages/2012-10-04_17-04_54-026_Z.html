There are eight types of chart: Area, Bar, Column, Gauge, Line, Pie, Radar and Scatter.

To explore these, see the Examples page in the Sencha documentation. 

var store = Ext.create('Ext.data.ArrayStore', {
    fields: ['name', 'data'],
    data: [['v1', 10], ['v2', 7], ['v3', 5], ['v4', 2], ['v5', 27]]
});
Ext.create('Ext.chart.Chart', {
    renderTo: Ext.getBody(),
    width: 500,
    height: 360,
    animate: true,
    store: store,
    series: [{
        type: 'pie',
        angleField: 'data'
    }]
});

var store = Ext.create('Ext.data.ArrayStore', {
    fields: ['data', 'name'],
    data: [
        [20, 'Does not resemble'],
        [80, 'Resembles']
    ]
});
Ext.create('Ext.chart.Chart', {
    store: store,
    series: [{
        type: 'pie',
        angleField: 'data',
        colorSet: ['#333', 'yellow'],
        tips: {
            trackMouse: true,
            width: 180,
            height: 28,
            renderer: function(record, item) {
                this.update(record.get('name') + ' Pac-Man.');
            }
        }
    }],
    renderTo: Ext.getBody(),
    width: 380,
    height: 380,
    style: {
        'background-color': '#000',
        padding: 12
    }
});
